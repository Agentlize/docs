## Import Alias

Sử dụng `@/` làm alias cho `src/`:

```typescript
// Thay vì:
import { Button } from '../../../components/ui/button';

// Dùng:
import { Button } from '@/components/ui/button';
```

Cấu hình trong `tsconfig.json`:

```json
{
  "compilerOptions": {
    "paths": {
      "@/*": ["./src/*"]
    }
  }
}
```

## Component Pattern

### Functional Component \+ Arrow Function

```typescript title="StatusBadge.tsx"
import type { StaffStatus } from '@/types';

interface StatusBadgeProps {
  status: StaffStatus;
}

export const StatusBadge = ({ status }: StatusBadgeProps) => {
  const config = {
    Pending: { label: 'Chờ duyệt', variant: 'secondary' },
    Approved: { label: 'Đã duyệt', variant: 'default' },
    Rejected: { label: 'Từ chối', variant: 'destructive' },
  };

  return <Badge variant={config[status].variant}>{config[status].label}</Badge>;
};
```

### Composition over Props Drilling

```typescript
// Thay vì:
<Table renderHeader={...} renderRow={...} renderFooter={...} />

// Dùng children:
<Table>
  <TableHeader>...</TableHeader>
  <TableBody>...</TableBody>
</Table>
```

## Styling

### Semantic Tokens Only

<Warning>
  **KHÔNG dùng** màu Tailwind trực tiếp (`bg-blue-500`, `text-red-600`).
</Warning>

**Chỉ dùng** semantic tokens:

```tsx
// ĐÚNG
<div className="bg-primary text-primary-foreground">...</div>
<div className="bg-muted text-muted-foreground">...</div>
<div className="bg-card border-border">...</div>

// SAI
<div className="bg-blue-500 text-white">...</div>
```

Tailwind config đã bật `darkMode: ['class']` — lớp `.dark` được `ThemeContext` gắn lên `<html>`.

### cn() Helper

```typescript title="src/lib/utils.ts"
import { clsx, type ClassValue } from 'clsx';
import { twMerge } from 'tailwind-merge';

export const cn = (...inputs: ClassValue[]) => {
  return twMerge(clsx(inputs));
};

// Sử dụng:
<button className={cn('px-4 py-2', isActive && 'bg-primary', isDisabled && 'opacity-50')} />
```

### Dialog Mobile Margin

Mọi `DialogContent` phải có:

```tsx
<DialogContent className="max-w-[calc(100%-2rem)] md:max-w-lg">
```

Đảm bảo dialog không full-width trên mobile.

## Toast

Dùng **sonner** (`import { toast } from 'sonner'`), KHÔNG dùng shadcn toast.

```typescript
import { toast } from 'sonner';

toast.success('Đã lưu thành công');
toast.error('Có lỗi xảy ra');
toast.warning('Cảnh báo');
toast.info('Thông tin');
```

## Type Safety

### Tránh `any`

```typescript
// SAI
const data: any = await fetch(...);

// ĐÚNG
const data: StaffProfile = await fetch(...);
```

### Dùng `unknown` thay thế khi cần

```typescript
const handleError = (error: unknown) => {
  if (error instanceof Error) {
    console.error(error.message);
  }
};
```

### Explicit return types

```typescript
// TỐT
export const getStaff = async (id: string): Promise<StaffProfile | null> => {
  return mockData.find(s => s.id === id) ?? null;
};
```
